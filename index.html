Question:1 Find the number of digits

            const Find_Digits = (N) => 
            {
                let count = 0;
                if (N >= 1) ++count;
                
                while (N / 10 >= 1) {
                    N /= 10;
                    ++count;
                }
                
                return count;
                
            };

Question:2 Find the Fives.

            const Find_Five = (N) =>
            {
                let count=0;
                while(N>0){
                    let z=N%10
                    if(z===5){
                        count++
                    }
                    N=parseInt(N/10)
                }
                return count
            };
